#сборка, состоит из нескольких подпроектов

#программа ждет от пользователя ввода двух чисел в консоль и выполнеят ариметические операции, только с использованием библиотеке Calc

#команды консоли для генерации и билда файлов проекта
#все действия в консоли начинаются с папки example3

mkdir build
cd build
cmake .. //генерация файлов проекта для IDE по умолчанию в каталоге build, если нужноо выбрать генератор, то cmake -G "название генератора".
# список генераторов можно глянуть cmake --help
cmake --build . //собираем исходник, попадет в папку src/Debug

# проверка и запуск
./src/Debug/MainCalc.exe
